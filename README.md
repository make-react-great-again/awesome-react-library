## Install

```bash
npm install --save awesome-react-library
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'awesome-react-library'
import 'awesome-react-library/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
